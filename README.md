Как запустить локальный сервер:

(у меня в моменте перестал отправлять с gmail, через яндекс нормально работает, но без впн)

export EMAIL_HOST="smtp.yandex.ru"

export EMAIL_PORT="465"

export EMAIL_USE_TLS="False"

export EMAIL_USE_SSL="True"
